#include "SmashHeroes.h"

// ==========================================
// SYSTEM 1: SQUAD SWAPPING (The "Tag" Mechanic)
// ==========================================

class USquadComponent : public UActorComponent
{
public:
    // The active squad of 3 heroes
    UPROPERTY(EditAnywhere)
    TArray<AHeroCharacter*> HeroSquad;

    // Index of currently controlled hero
    int32 CurrentHeroIndex;

    void SwapHero(int32 NewIndex)
    {
        if (!HeroSquad.IsValidIndex(NewIndex) || NewIndex == CurrentHeroIndex) return;

        // 1. Get current and new hero
        AHeroCharacter* OldHero = HeroSquad[CurrentHeroIndex];
        AHeroCharacter* NewHero = HeroSquad[NewIndex];

        // 2. Save state of old hero (Health, Energy, Cooldowns)
        OldHero->SaveHeroState();

        // 3. VFX: Spawn "Tag out" particle effect at OldHero location
        SpawnVFX(TagOutEffect, OldHero->GetActorLocation());

        // 4. Teleport New Hero to Old Hero's location
        NewHero->SetActorLocation(OldHero->GetActorLocation());
        NewHero->SetActorRotation(OldHero->GetActorRotation());

        // 5. Controller Possession (Seamless transfer of input)
        APlayerController* PC = Cast<APlayerController>(GetOwner());
        if (PC)
        {
            PC->Possess(NewHero);
        }

Duo Synergy: Certain hero pairs unlock unique "Synergy Moves." (e.g., Daddy throws his cap, and Linken can shoot an arrow through it to create a homing fireball).

Unlock Method: Transformations are earned through completing a hero‚Äôs personal "Transformation Quest," a difficult, story-driven trial that forces them to confront their past.

Multi-Phase Fights: A boss might start as a slow, armored brute, then shatter its armor to become a fast, agile attacker in phase two.

Facilities: A central hub you build and upgrade. Construct a Training Hall to spar and test builds, a Tech Lab (run by Tails and Dr. Light) to research new gear, a War Room to launch missions, and Warp Gates for fast travel.
